# API Design Best Practices

Creating well-designed APIs is essential for building maintainable and scalable applications. This guide covers REST, GraphQL, and gRPC design patterns.

## REST API Design

### Resource Naming

Use clear, consistent resource names:

```
# Good
GET    /api/v1/users
GET    /api/v1/users/{id}
POST   /api/v1/users
PUT    /api/v1/users/{id}
PATCH  /api/v1/users/{id}
DELETE /api/v1/users/{id}

# Nested resources
GET    /api/v1/users/{id}/orders
POST   /api/v1/users/{id}/orders
GET    /api/v1/users/{id}/orders/{orderId}

# Bad
GET    /api/v1/getAllUsers
GET    /api/v1/user_list
POST   /api/v1/createNewUser
```

### HTTP Status Codes

Use appropriate status codes:

```typescript
enum HttpStatus {
  // Success
  OK = 200,
  CREATED = 201,
  ACCEPTED = 202,
  NO_CONTENT = 204,

  // Redirection
  MOVED_PERMANENTLY = 301,

  NOT_MODIFIED = 304,

  // Client Errors
  BAD_REQUEST = 400,

  FORBIDDEN = 403,
  NOT_FOUND = 404,
  METHOD_NOT_ALLOWED = 405,
  CONFLICT = 409,
  UNPROCESSABLE_ENTITY = 422,
  TOO_MANY_REQUESTS = 429,

  // Server Errors
  INTERNAL_SERVER_ERROR = 500,
  NOT_IMPLEMENTED = 501,
  BAD_GATEWAY = 502,
  SERVICE_UNAVAILABLE = 503,
  GATEWAY_TIMEOUT = 504
}

// Example usage in Express
app.post('/api/users', async (req, res) => {
  try {
    const user = await createUser(req.body);
    res.status(HttpStatus.CREATED).json(user);
  } catch (error) {
    if (error.code === 'DUPLICATE_EMAIL') {
      res.status(HttpStatus.CONFLICT).json({
        error: 'Email already exists',
        code: 'DUPLICATE_EMAIL'
      });
    } else {
      res.status(HttpStatus.INTERNAL_SERVER_ERROR).json({
        error: 'Internal server error',
        code: 'INTERNAL_ERROR'
      });
    }
  }
});
```

### Pagination

Implement cursor-based pagination for large datasets:

```python
from typing import Optional, List, Dict, Any
from dataclasses import dataclass

@dataclass
class PaginationParams:
    cursor: Optional[str] = None
    limit: int = 20

@dataclass
class PaginatedResponse:
    data: List[Dict[str, Any]]
    next_cursor: Optional[str]
    has_more: bool
    total_count: Optional[int] = None

def paginate_users(params: PaginationParams) -> PaginatedResponse:
    query = db.users.find()

    if params.cursor:
        # Decode cursor and apply filter
        cursor_data = decode_cursor(params.cursor)
        query = query.where('id > ?', cursor_data['last_id'])

    # Fetch one extra to check if there are more results
    users = query.limit(params.limit + 1).execute()

    has_more = len(users) > params.limit
    if has_more:
        users = users[:params.limit]

    next_cursor = None
    if has_more:
        next_cursor = encode_cursor({
            'last_id': users[-1]['id']
        })

    return PaginatedResponse(
        data=users,
        next_cursor=next_cursor,
        has_more=has_more
    )

# API endpoint
@app.get('/api/users')
def get_users(cursor: Optional[str] = None, limit: int = 20):
    params = PaginationParams(cursor=cursor, limit=min(limit, 100))
    result = paginate_users(params)

    return {
        'data': result.data,
        'pagination': {
            'next_cursor': result.next_cursor,
            'has_more': result.has_more,
            'limit': params.limit
        }
    }
```

### Filtering and Sorting

```ruby
class UsersController < ApplicationController
  def index
    users = User.all

    # Filtering
    users = users.where(status: params[:status]) if params[:status]
    users = users.where('created_at >= ?', params[:from_date]) if params[:from_date]
    users = users.where('age >= ?', params[:min_age]) if params[:min_age]

    # Searching
    if params[:search]
      users = users.where(
        'name ILIKE ? OR email ILIKE ?',
        "%#{params[:search]}%",
        "%#{params[:search]}%"
      )
    end

    # Sorting
    sort_field = params[:sort_by] || 'created_at'
    sort_order = params[:order]&.downcase == 'asc' ? 'ASC' : 'DESC'

    allowed_fields = %w[name email created_at age]
    if allowed_fields.include?(sort_field)
      users = users.order("#{sort_field} #{sort_order}")
    end

    # Pagination
    page = (params[:page] || 1).to_i
    per_page = [(params[:per_page] || 20).to_i, 100].min

    users = users.page(page).per(per_page)

    render json: {
      data: users,
      meta: {
        current_page: page,
        per_page: per_page,
        total_pages: users.total_pages,
        total_count: users.total_count
      }
    }
  end
end
```

## GraphQL API Design

### Schema Design

```graphql
type Query {
  user(id: ID!): User

    first: Int = 20
    after: String

    orderBy: UserOrderBy
  ): UserConnection!

type Mutation {
  createUser(input: CreateUserInput!): CreateUserPayload!
  updateUser(id: ID!, input: UpdateUserInput!): UpdateUserPayload!
  deleteUser(id: ID!): DeleteUserPayload!

  publishPost(id: ID!): PublishPostPayload!
}

type User {
  id: ID!
  name: String!
  email: String!
  avatar: String
  bio: String
  posts(first: Int = 10, after: String): PostConnection!
  followers(first: Int = 10, after: String): UserConnection!
  following(first: Int = 10, after: String): UserConnection!
  createdAt: DateTime!
  updatedAt: DateTime!
}

type Post {
  id: ID!
  title: String!
  content: String!
  excerpt: String
  author: User!
  tags: [String!]!
  status: PostStatus!
  publishedAt: DateTime
  createdAt: DateTime!
  updatedAt: DateTime!
}

enum PostStatus {
  DRAFT
  PUBLISHED
  ARCHIVED
}

input UserFilter {
  status: UserStatus
  role: UserRole
  createdAfter: DateTime
  createdBefore: DateTime
  search: String
}

input UserOrderBy {
  field: UserOrderField!
  direction: OrderDirection!
}

enum UserOrderField {
  NAME
  EMAIL
  CREATED_AT
  UPDATED_AT
}

  ASC
  DESC
}

type UserConnection {
  edges: [UserEdge!]!
  pageInfo: PageInfo!
  totalCount: Int!
}

type UserEdge {
  node: User!
  cursor: String!
}

type PageInfo {

  hasPreviousPage: Boolean!
  startCursor: String
  endCursor: String
```

### Resolver Implementation

```javascript
const resolvers = {
  Query: {
    user: async (_, { id }, context) => {
      return context.dataloaders.userLoader.load(id);
    },

    users: async (_, { first, after, filter, orderBy }, context) => {
      const result = await context.db.users.findMany({
        where: buildWhereClause(filter),
        orderBy: buildOrderByClause(orderBy),
        take: first + 1,
        skip: after ? 1 : 0,
        cursor: after ? { id: after } : undefined
      });

      const hasNextPage = result.length > first;
      const edges = result.slice(0, first).map(user => ({
        node: user,
        cursor: user.id
      }));

      return {
        edges,
        pageInfo: {
          hasNextPage,
          hasPreviousPage: !!after,
          startCursor: edges[0]?.cursor,
          endCursor: edges[edges.length - 1]?.cursor
        },
        totalCount: await context.db.users.count({
          where: buildWhereClause(filter)
        })
      };
    }
  },

  Mutation: {
    createUser: async (_, { input }, context) => {
      // Validate input
      const errors = validateUserInput(input);
      if (errors.length > 0) {
        return {
          success: false,
          errors,
          user: null
        };
      }

      // Create user
      const user = await context.db.users.create({
        data: input
      });

      // Clear cache
      await context.cache.delete(`user:${user.id}`);

      return {
        success: true,
        errors: [],
        user
      };
    }
  },

  User: {
    posts: async (user, { first, after }, context) => {
      return context.dataloaders.userPostsLoader.load({
        userId: user.id,
        first,
        after
      });
    },

    followers: async (user, args, context) => {
      return context.dataloaders.userFollowersLoader.load({
        userId: user.id,
        ...args
      });
    }
  }
};
```

### DataLoader for N+1 Prevention

```typescript
import DataLoader from 'dataloader';

class Dataloaders {
  userLoader: DataLoader<string, User>;
  postLoader: DataLoader<string, Post>;
  userPostsLoader: DataLoader<string, Post[]>;

  constructor(db: Database) {
    this.userLoader = new DataLoader(async (ids) => {
      const users = await db.users.findMany({
        where: { id: { in: ids as string[] } }
      });

      const userMap = new Map(users.map(u => [u.id, u]));
      return ids.map(id => userMap.get(id as string));
    });

    this.postLoader = new DataLoader(async (ids) => {
      const posts = await db.posts.findMany({
        where: { id: { in: ids as string[] } }
      });

      const postMap = new Map(posts.map(p => [p.id, p]));
      return ids.map(id => postMap.get(id as string));
    });

    this.userPostsLoader = new DataLoader(async (userIds) => {
      const posts = await db.posts.findMany({
        where: { authorId: { in: userIds as string[] } },
        orderBy: { createdAt: 'desc' }
      });

      const postsByUser = new Map<string, Post[]>();
      for (const post of posts) {
        if (!postsByUser.has(post.authorId)) {
          postsByUser.set(post.authorId, []);
        }
        postsByUser.get(post.authorId)!.push(post);
      }

      return userIds.map(id => postsByUser.get(id as string) || []);
    });
  }
}
```

## Rate Limiting

```go
package middleware

import (
    "net/http"
    "sync"
    "time"

    "golang.org/x/time/rate"
)

type RateLimiter struct {
    limiters map[string]*rate.Limiter
    mu       sync.RWMutex
    rate     rate.Limit
    burst    int
}

func NewRateLimiter(r rate.Limit, b int) *RateLimiter {
    return &RateLimiter{
        limiters: make(map[string]*rate.Limiter),
        rate:     r,
        burst:    b,
    }
}

func (rl *RateLimiter) getLimiter(key string) *rate.Limiter {
    rl.mu.RLock()
    limiter, exists := rl.limiters[key]
    rl.mu.RUnlock()

    if !exists {
        rl.mu.Lock()
        limiter = rate.NewLimiter(rl.rate, rl.burst)
        rl.limiters[key] = limiter
        rl.mu.Unlock()
    }

    return limiter
}

func (rl *RateLimiter) Middleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        // Use IP address as the key
        key := r.RemoteAddr

        // Get or create limiter for this key
        limiter := rl.getLimiter(key)

        if !limiter.Allow() {
            w.Header().Set("X-RateLimit-Limit", fmt.Sprintf("%d", rl.burst))
            w.Header().Set("X-RateLimit-Remaining", "0")
            w.Header().Set("Retry-After", "60")

            http.Error(w, "Rate limit exceeded", http.StatusTooManyRequests)
            return
        }

        next.ServeHTTP(w, r)
    })
}

// Usage
func main() {
    mux := http.NewServeMux()
    mux.HandleFunc("/api/", apiHandler)

    // Allow 10 requests per second with burst of 20
    rateLimiter := NewRateLimiter(10, 20)

    http.ListenAndServe(":8080", rateLimiter.Middleware(mux))
}
```

</Tip>
