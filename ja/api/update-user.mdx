## パスパラメータ

<ParamField path="id" type="string" required>
  更新するユーザーの一意の識別子
</ParamField>

## ボディパラメータ

<ParamField body="name" type="string">
  ユーザーのフルネーム
</ParamField>

<ParamField body="email" type="string">
  ユーザーのメールアドレス（一意である必要があります）
</ParamField>

<ParamField body="bio" type="string">
  ユーザーの自己紹介
</ParamField>

<ParamField body="avatar_url" type="string">
  プロフィール画像のURL
</ParamField>

<ParamField body="preferences" type="object">
  ユーザー設定オブジェクト
  <Expandable title="プロパティ">
    <ParamField body="theme" type="string">
      テーマ設定: `light`、`dark`、または`auto`
    </ParamField>
    <ParamField body="language" type="string">
      優先言語コード（例: `ja`、`en`、`es`）
    </ParamField>
    <ParamField body="notifications" type="boolean">
      通知の有効/無効
    </ParamField>
    <ParamField body="timezone" type="string">
      タイムゾーン（例: `Asia/Tokyo`）
    </ParamField>
  </Expandable>
</ParamField>

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "net/http"
)

type UserUpdate struct {
    Name        string      `json:"name"`
    Bio         string      `json:"bio"`
    Preferences Preferences `json:"preferences"`
}

type Preferences struct {
    Theme         string `json:"theme"`
    Language      string `json:"language"`
    Notifications bool   `json:"notifications"`
    Timezone      string `json:"timezone"`
}

func main() {
    url := "https://api.example.com/api/v1/users/user_123"

    data := UserUpdate{
        Name: "田中花子",
        Bio:  "大阪在住のフルスタックエンジニア",
        Preferences: Preferences{
            Theme:         "dark",
            Language:      "ja",
            Notifications: true,
            Timezone:      "Asia/Tokyo",
        },
    }

    jsonData, _ := json.Marshal(data)
    req, _ := http.NewRequest("PUT", url, bytes.NewBuffer(jsonData))
    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, _ := client.Do(req)
    defer resp.Body.Close()
}
```
