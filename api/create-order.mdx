---
title: "Create Order"
api: "POST /api/v1/orders"
description: "Create a new order with items and shipping information"
---

## Body Parameters

<ParamField body="user_id" type="string" required>
  The ID of the user placing the order
</ParamField>

<ParamField body="items" type="array" required>
  Array of order items
  <Expandable title="item object">
    <ParamField body="product_id" type="string" required>
      Product ID
    </ParamField>
    <ParamField body="quantity" type="integer" required>
      Quantity to order
    </ParamField>
    <ParamField body="price" type="number" required>
      Price per unit
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="shipping_address" type="object" required>
  Shipping address details
  <Expandable title="properties">
    <ParamField body="street" type="string" required>
      Street address
    </ParamField>
    <ParamField body="city" type="string" required>
      City
    </ParamField>
    <ParamField body="state" type="string">
      State or province
    </ParamField>
    <ParamField body="zip" type="string" required>
      ZIP or postal code
    </ParamField>
    <ParamField body="country" type="string" required>
      Country code (ISO 3166-1 alpha-2)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="payment_method_id" type="string" required>
  Payment method identifier
</ParamField>

<ParamField body="notes" type="string">
  Optional order notes
</ParamField>

```python Python
import requests

url = "https://api.example.com/api/v1/orders"
headers = {
    "Authorization": "Bearer YOUR_API_KEY",
    "Content-Type": "application/json"
}
data = {
    "user_id": "user_456",
    "items": [
        {
            "product_id": "prod_789",
            "quantity": 2,
            "price": 49.99
        }
    ],
    "shipping_address": {
        "street": "123 Main St",
        "city": "New York",
        "state": "NY",
        "zip": "10001",
        "country": "US"
    },
    "payment_method_id": "pm_123"
}

response = requests.post(url, json=data, headers=headers)
print(response.json())
```
